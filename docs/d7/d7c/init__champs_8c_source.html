82 {
83  int fd = 0; int size = 0; int i = 0; char c = 0;
84  if (!prog || !arena) return false;
85  if (!prog->prog_name) return false;
86  if ((fd = open(prog->prog_name, O_RDONLY)) == -1) return false;
87  if ((size = lseek(fd, 0, SEEK_END)) == -1) return false;
88  if (size < FULL_HEADER_SIZE) return false;
