81bool load_champ(params_progs_t *prog, byte_t *arena)
82{
83 int fd = 0; int size = 0; int i = 0; char c = 0;
84 if (!prog || !arena) return false;
85 if (!prog->prog_name) return false;
86 if ((fd = open(prog->prog_name, O_RDONLY)) == -1) return false;
87 if ((size = lseek(fd, 0, SEEK_END)) == -1) return false;
88 if (size < FULL_HEADER_SIZE) return false;
89 if (lseek(fd, FULL_HEADER_SIZE, SEEK_SET) == -1) return false;
90 for (i = 0; i < size - FULL_HEADER_SIZE; i++) {
91 if (read(fd, &c, 1) == -1) return false;
92 arena[get_arena_adress(prog->load_address + i)] = c;
93 }
94 if (close(fd) == -1) return false;
95 return true;
96}

122 return false;
123 }
124 }
