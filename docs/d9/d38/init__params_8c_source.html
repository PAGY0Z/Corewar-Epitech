184params_t *init_params(char **argv, int mall_v1, int mall_v2)
185{
186 params_t *params = NULL; success_t dump_return = NONE;
187 if (!argv) return NULL;

189 params->dump_nbr_cycles = -1;

191 params->progs = NULL;
192 dump_return = get_dump_nbr_cycle(argv, params);
193 if (dump_return == FAILURE) return e_init_params(params);
194 if (dump_return == SUCCESS) argv += 2;
195 if (!(get_params_prog_nbr(argv, params))) return e_init_params(params);
196 if (!(init_params_progs(params, mall_v2))) return e_init_params(params);

198 my_putstr("Error : Invalid program parameters\n", 2);
